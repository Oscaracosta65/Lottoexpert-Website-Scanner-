function leAuditNormalizeUrl(string $url, array $config): string
{
    $url = html_entity_decode(trim($url), ENT_QUOTES, 'UTF-8');

    if ($url === '') {
        return '';
    }

        $__leForeachTmp1 = $config['ignore_patterns'];
    $__leForeachKeys1 = is_array($__leForeachTmp1) ? array_keys($__leForeachTmp1) : [];

        $pattern = $__leForeachTmp1[$__leForeachKey1];

            return '';
        }
        }

    if (strpos($url, '//') === 0) {
        $url = 'https:' . $url;
    }

    if (strpos($url, '/') === 0) {
        $url = rtrim($config['site_root'], '/') . $url;
    }

    if (!preg_match('#^https?://#i', $url)) {
        return '';
    }

    $parts = parse_url($url);

    if (empty($parts['host'])) {
        return '';
    }

    $host = strtolower($parts['host']);

    if (!in_array($host, $config['allowed_hosts'], true)) {
        return '';
    }

    $scheme = 'https';
    $path   = $parts['path'] ?? '/';
    $query  = $parts['query'] ?? '';

    $path = preg_replace('#/+#', '/', $path);

    $clean = $scheme . '://' . $config['allowed_host'] . $path;

    if ($query !== '') {
        parse_str($query, $queryArray);

        $blockedQueryKeys = [
            'utm_source', 'utm_medium', 'utm_campaign', 'utm_term', 'utm_content',
            'fbclid', 'gclid', 'msclkid', 'print', 'tmpl', 'format',
        ];

                $__leForeachTmp2 = $blockedQueryKeys;
        $__leForeachKeys2 = is_array($__leForeachTmp2) ? array_keys($__leForeachTmp2) : [];
        for ($__leForeachIndex2 = 0, $__leForeachCount2 = count($__leForeachKeys2); $__leForeachIndex2 < $__leForeachCount2; $__leForeachIndex2++) {

function leAuditFetchUrl(string $url, array $config, ?int $timeout = null): array
{
    $timeoutSeconds = $timeout ?? (int) $config['request_timeout'];
    $response = [
        'url'              => $url,
        'status'           => 0,
        'final_url'        => $url,
        'content_type'     => '',
        'body'             => '',
        'error'            => '',
        'load_time_ms'     => 0,
        'response_headers' => '',
    ];

    $start = microtime(true);

    if (!function_exists('curl_init')) {
        $response['error'] = 'cURL is not available on this server.';
        return $response;
    }

    $ch               = curl_init();
    $headersCollected = '';

    curl_setopt_array($ch, [
        CURLOPT_URL            => $url,
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_FOLLOWLOCATION => true,
        CURLOPT_MAXREDIRS      => 5,
        CURLOPT_CONNECTTIMEOUT => $timeoutSeconds,
        CURLOPT_TIMEOUT        => $timeoutSeconds,
        CURLOPT_SSL_VERIFYPEER => true,
        CURLOPT_SSL_VERIFYHOST => 2,
        CURLOPT_USERAGENT      => 'LottoExpert Private Audit Scanner/2.0',
        CURLOPT_HEADER         => false,
        CURLOPT_ENCODING       => '',
        CURLOPT_HEADERFUNCTION => function ($ch, $header) use (&$headersCollected) {
            $headersCollected .= $header;
            return strlen($header);
        },
    ]);

    $body = curl_exec($ch);

    if ($body === false) {
        $response['error'] = curl_error($ch);
    } else {
        $response['body'] = (string) $body;
    }

    $response['status']           = (int) curl_getinfo($ch, CURLINFO_HTTP_CODE);
    $response['final_url']        = (string) curl_getinfo($ch, CURLINFO_EFFECTIVE_URL);
    $response['content_type']     = (string) curl_getinfo($ch, CURLINFO_CONTENT_TYPE);
    $response['load_time_ms']     = (int) round((microtime(true) - $start) * 1000);
    $response['response_headers'] = $headersCollected;

    curl_close($ch);
    return $response;
}

function leAuditCompactText(string $text, int $maxLength = 280): string
{
    $text = html_entity_decode(strip_tags($text), ENT_QUOTES, 'UTF-8');
    $text = preg_replace('/\s+/u', ' ', $text);
    $text = trim((string) $text);

    if ($text === '') {
        return '';
    }

    if (mb_strlen($text, 'UTF-8') <= $maxLength) {
        return $text;
    }

    return mb_substr($text, 0, $maxLength - 3, 'UTF-8') . '...';
}

function leAuditExtractAttr(string $tag, string $attr): string
{

    }
    return '';
}

function leAuditBuildSourceContext(string $html, int $offset, int $length): array
{
    $beforeStart = max(0, $offset - 300);
    $afterEnd    = min(strlen($html), $offset + $length + 300);
    $rawContext  = substr($html, $beforeStart, $afterEnd - $beforeStart);
    $rawSnippet  = substr($html, $offset, $length);

    return [
        'source_excerpt' => leAuditCompactText($rawContext, 420),
        'html_snippet'   => leAuditCompactText($rawSnippet, 420),

function leAuditResolveImageUrl(string $src, string $baseUrl, array $config): string
{
    $src = html_entity_decode(trim($src), ENT_QUOTES, 'UTF-8');

    if ($src === '' || strpos($src, 'data:') === 0) {
        return '';
    }

    if (strpos($src, '//') === 0) {
        $src = 'https:' . $src;
    } elseif (strpos($src, '/') === 0) {
        $src = rtrim($config['site_root'], '/') . $src;
    } elseif (!preg_match('#^https?://#i', $src)) {
        $baseParts = parse_url($baseUrl);
        $baseRoot  = 'https://' . ($baseParts['host'] ?? $config['allowed_host']);
        $basePath  = isset($baseParts['path']) ? dirname($baseParts['path']) : '';
        $src       = rtrim($baseRoot . '/' . trim($basePath, '/'), '/') . '/' . ltrim($src, '/');
    }

    return preg_match('#^https?://#i', $src) ? $src : '';
}

function leAuditExtractLinkContexts(string $html, string $baseUrl, array $config): array
{
    $contexts = [];

    if (!preg_match_all('#<a\s[^>]*href=["\']([^"\']+)["\'][^>]*>(.*?)</a>#is', $html, $matches, PREG_SET_ORDER | PREG_OFFSET_CAPTURE)) {
        return $contexts;
    }

        $__leForeachTmp3 = $matches;
    $__leForeachKeys3 = is_array($__leForeachTmp3) ? array_keys($__leForeachTmp3) : [];

        $__leForeachKey3 = $__leForeachKeys3[$__leForeachIndex3];

        $tag      = $match[0][0];
        $offset   = (int) $match[0][1];
        $href     = $match[1][0];
        $inner    = $match[2][0] ?? '';
        $internal = leAuditNormalizeUrl($href, $config);
        $external = $internal === '' ? leAuditNormalizeExternalUrlForReport($href, $config) : '';
        $key      = $internal !== '' ? $internal : $external;

        if ($key === '') {
            continue;
        }

        if (isset($contexts[$key])) {

        $context = leAuditBuildSourceContext($html, $offset, strlen($tag));
        $context['link_text'] = leAuditCompactText($inner, 220);

        $contexts[$key]       = $context;
        }

    return $contexts;
}

function leAuditHeadCheckUrl(string $url, int $timeout = 5): array
{
    if (!function_exists('curl_init')) {
        return ['status' => 0, 'final_url' => $url, 'method' => 'HEAD', 'error' => 'cURL unavailable'];
    }

    // Try HEAD first
    $ch = curl_init();
    curl_setopt_array($ch, [
        CURLOPT_URL            => $url,
        CURLOPT_NOBODY         => true,
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_FOLLOWLOCATION => true,
        CURLOPT_MAXREDIRS      => 5,
        CURLOPT_CONNECTTIMEOUT => $timeout,
        CURLOPT_TIMEOUT        => $timeout,
        CURLOPT_SSL_VERIFYPEER => false,
        CURLOPT_SSL_VERIFYHOST => 0,
        CURLOPT_USERAGENT      => 'LottoExpert Private Audit Scanner/3.0',
    ]);
    curl_exec($ch);
    $headStatus  = (int)    curl_getinfo($ch, CURLINFO_HTTP_CODE);
    $headFinalUrl = (string) curl_getinfo($ch, CURLINFO_EFFECTIVE_URL);
    $headError   = curl_error($ch);
    curl_close($ch);

    // Retry with GET for ambiguous or blocked statuses
    $retryOnStatus = [0, 403, 405, 429];
    if (in_array($headStatus, $retryOnStatus, true) || $headStatus >= 500) {
        $ch2 = curl_init();
        curl_setopt_array($ch2, [
            CURLOPT_URL            => $url,
            CURLOPT_RETURNTRANSFER => true,
            CURLOPT_FOLLOWLOCATION => true,
            CURLOPT_MAXREDIRS      => 5,
            CURLOPT_CONNECTTIMEOUT => $timeout,
            CURLOPT_TIMEOUT        => $timeout,
            CURLOPT_SSL_VERIFYPEER => false,
            CURLOPT_SSL_VERIFYHOST => 0,
            CURLOPT_USERAGENT      => 'LottoExpert Private Audit Scanner/3.0',
            CURLOPT_ENCODING       => '',
        ]);
        curl_exec($ch2);
        $getStatus   = (int)    curl_getinfo($ch2, CURLINFO_HTTP_CODE);
        $getFinalUrl = (string) curl_getinfo($ch2, CURLINFO_EFFECTIVE_URL);
        $getError    = curl_error($ch2);
        curl_close($ch2);
        if ($getStatus > 0) {
            return ['status' => $getStatus, 'final_url' => $getFinalUrl, 'method' => 'GET', 'error' => $getError];
        }
    }

    return ['status' => $headStatus, 'final_url' => $headFinalUrl, 'method' => 'HEAD', 'error' => $headError];
}

function leAuditCountH1(string $html): int
{
    return preg_match_all('#<h1\b[^>]*>(.*?)</h1>#is', $html) ?: 0;
}

function leAuditExtractH1Text(string $html): string
{
    if (preg_match('#<h1\b[^>]*>(.*?)</h1>#is', $html, $m)) {
        return trim(html_entity_decode(strip_tags($m[1]), ENT_QUOTES, 'UTF-8'));
    }
    return '';
}

function leAuditHasNoindex(string $html, string $responseHeaders = ''): bool
{
    if (preg_match('#<meta[^>]+name=["\']robots["\'][^>]+content=["\'][^"\']*noindex[^"\']*["\']#i', $html)) {
        return true;
    }
    if ($responseHeaders !== '' && preg_match('#x-robots-tag:[^\r\n]*noindex#i', $responseHeaders)) {
        return true;
    }
    return false;
}

function leAuditCheckUrl(string $url, array $config, array $options = []): array
{
    $checkSeo            = $options['check_seo']             ?? true;
    $checkSpeed          = $options['check_speed']           ?? true;
    $checkAltText        = $options['check_alt_text']        ?? true;
    $checkBrokenInternal = $options['check_broken_internal'] ?? true;

    $fetch = leAuditFetchUrl($url, $config);
    $html  = $fetch['body'];

    $title       = $checkSeo     ? leAuditExtractTag($html, 'title')          : '';
    $description = $checkSeo     ? leAuditExtractMetaDescription($html)        : '';
    $canonical   = $checkSeo     ? leAuditExtractCanonical($html)              : '';
    $h1Count     = $checkSeo     ? leAuditCountH1($html)                       : 0;
    $h1Text      = $checkSeo     ? leAuditExtractH1Text($html)                 : '';
    $noindex     = $checkSeo     ? leAuditHasNoindex($html, $fetch['response_headers']) : false;
    $imgsNoAlt   = $checkAltText ? leAuditCountImagesWithoutAlt($html)         : 0;

    $titleLen = mb_strlen($title, 'UTF-8');
    $descLen  = mb_strlen($description, 'UTF-8');

    $issues   = [];
    $warnings = [];

    if ($fetch['status'] === 0) {
        // Always flag complete connection failures - these indicate the URL is unreachable
        $issues[] = 'Connection failed: ' . $fetch['error'];
    } elseif ($checkBrokenInternal) {
        // Only flag 4xx/5xx as issues when the "Broken Internal Pages" check is enabled
        if ($fetch['status'] === 404) {
            $issues[] = '404 Not Found';
        } elseif ($fetch['status'] >= 500) {
            $issues[] = 'Server error ' . $fetch['status'];
        } elseif ($fetch['status'] >= 400) {
            $issues[] = 'Client error ' . $fetch['status'];
        }
    }

    if ($checkSeo && $fetch['status'] >= 200 && $fetch['status'] < 300) {
        if ($title === '') {
            $issues[] = 'Missing title tag';
        } elseif ($titleLen < 50) {
            $warnings[] = 'Page title is too short (' . $titleLen . ' chars) - Google recommends 50-70 characters. A very short title gives search engines almost no information about the page topic. Expand your title to clearly describe the page content.';
        } elseif ($titleLen > 70) {
            $warnings[] = 'Page title is too long (' . $titleLen . ' chars) - Google typically truncates titles longer than 70 characters in search results, cutting off part of your message. Edit the title to 50-70 characters.';
        }

        if ($description === '') {
            $issues[] = 'Missing meta description';
        } elseif ($descLen < 50) {
            $warnings[] = 'Meta description is too short (' . $descLen . ' chars) - Search engines use this text as the snippet shown under your page title in results. A snippet under 50 characters is too brief to attract clicks. Write a 50-165 character description that summarises the page and encourages users to click.';
        } elseif ($descLen > 165) {
            $warnings[] = 'Meta description is too long (' . $descLen . ' chars) - Google will cut off descriptions longer than ~165 characters, leaving an incomplete sentence in search results. Trim your description to under 165 characters.';
        }

        if ($canonical === '') {
            $warnings[] = 'Missing canonical tag - A canonical tag tells search engines which version of this URL is the official one, preventing duplicate-content penalties when the same page is reachable via multiple URLs. Add a self-referencing canonical tag to the head of this page.';
        }

        if ($h1Count === 0) {
            $issues[] = 'Missing H1 tag';
        } elseif ($h1Count > 1) {
            $warnings[] = 'Multiple H1 headings found (' . $h1Count . ') - Each page should have exactly one H1 tag that clearly states the main topic. Having several H1s confuses search engines about the primary subject. Keep one H1 and use H2/H3 for sub-headings.';
        }

        if ($noindex) {
            $warnings[] = 'Page has a noindex directive - This page is actively instructing search engines NOT to index it. If this is unintentional, remove the noindex directive so the page can appear in search results.';
        }
    }

    if ($checkAltText && $imgsNoAlt > 0 && $fetch['status'] >= 200 && $fetch['status'] < 300) {
        $warnings[] = $imgsNoAlt . ' image(s) are missing alt text - The alt attribute on an img tag describes the image to search engines and screen-reader users. Missing alt text harms SEO and accessibility. Add a short, descriptive alt attribute to every image on this page.';
    }

    if ($checkSpeed && $fetch['load_time_ms'] > 3000 && $fetch['status'] >= 200 && $fetch['status'] < 300) {
        $warnings[] = 'Slow page load time (' . $fetch['load_time_ms'] . ' ms) - Google uses page speed as a ranking factor; pages taking over 3 seconds to load are penalised and have higher visitor bounce rates. Investigate large images, render-blocking scripts, or slow server response times.';
    }

    // -- Performance diagnostics ----------------------------------------------
    $htmlSizeKb       = round(strlen($html) / 1024, 1);
    $scriptCount      = substr_count(strtolower($html), '<script');
    $stylesheetCount  = substr_count(strtolower($html), '<link') + substr_count(strtolower($html), '<style');
    $imageCount       = substr_count(strtolower($html), '<img');
    $inlineScriptCount = 0;
    if (preg_match_all('#<script(?![^>]*\bsrc\b)[^>]*>#i', $html, $inlineScripts)) {
        $inlineScriptCount = count($inlineScripts[0]);
    }
    $thirdPartyDomains = [];
    if (preg_match_all('#(?:src|href)=["\']https?://([^/"\']+)#i', $html, $domainMatches)) {
                $__leForeachTmp9 = $domainMatches[1];
        $__leForeachKeys9 = is_array($__leForeachTmp9) ? array_keys($__leForeachTmp9) : [];
        for ($__leForeachIndex9 = 0, $__leForeachCount9 = count($__leForeachKeys9); $__leForeachIndex9 < $__leForeachCount9; $__leForeachIndex9++) {
            $__leForeachKey9 = $__leForeachKeys9[$__leForeachIndex9];
            $d = $__leForeachTmp9[$__leForeachKey9];
            $d = strtolower($d);
            if (!in_array($d, $config['allowed_hosts'], true)) {
                $thirdPartyDomains[$d] = true;
            }
                }
    }
    $thirdPartyDomainCount = count($thirdPartyDomains);
    $slowTier = 'Fast';
    if ($fetch['load_time_ms'] > 5000)      { $slowTier = 'Very Slow'; }
    elseif ($fetch['load_time_ms'] > 3000)  { $slowTier = 'Slow'; }
    elseif ($fetch['load_time_ms'] > 1000)  { $slowTier = 'Acceptable'; }

    // -- Content quality ------------------------------------------------------
    $bodyText    = preg_replace('#<(script|style|noscript)[^>]*>.*?</\1>#is', '', $html);
    $bodyText    = preg_replace('/\s+/', ' ', trim(strip_tags($bodyText)));
    $wordCount   = ($bodyText !== '') ? count(explode(' ', $bodyText)) : 0;
    $thinContent = ($wordCount < 250 && $fetch['status'] >= 200 && $fetch['status'] < 300);
    $h2Count     = (int) (preg_match_all('#<h2\b#i', $html) ?: 0);
    $h3Count     = (int) (preg_match_all('#<h3\b#i', $html) ?: 0);
    $paragraphCount = (int) (preg_match_all('#<p\b#i', $html) ?: 0);

    // -- Structured data (JSON-LD) --------------------------------------------
    $jsonLdCount = 0; $invalidJsonLdCount = 0; $schemaTypes = [];
    if (preg_match_all('#<script[^>]+type=["\']application/ld\+json["\'][^>]*>(.*?)</script>#is', $html, $jldMatches)) {
                $__leForeachTmp10 = $jldMatches[1];
        $__leForeachKeys10 = is_array($__leForeachTmp10) ? array_keys($__leForeachTmp10) : [];
        for ($__leForeachIndex10 = 0, $__leForeachCount10 = count($__leForeachKeys10); $__leForeachIndex10 < $__leForeachCount10; $__leForeachIndex10++) {
            $__leForeachKey10 = $__leForeachKeys10[$__leForeachIndex10];
            $jldRaw = $__leForeachTmp10[$__leForeachKey10];
            $decoded = json_decode(trim($jldRaw), true);
            if ($decoded !== null) {
                $jsonLdCount++;
                $t = $decoded['@type'] ?? ($decoded[0]['@type'] ?? '');
                if ($t !== '') { $schemaTypes[] = (string) $t; }
            } else {
                $invalidJsonLdCount++;
            }
                }
    }

    // -- Page type and importance ---------------------------------------------
    $pageType       = leAuditDetectPageType($url, $title, $h1Text);
    $importance     = leAuditDetectImportance($url, $pageType);
    $isImportant    = $importance['is_important'];
    $importanceScore = $importance['importance_score'];

    // -- Indexability ---------------------------------------------------------
    $statusIndexable   = ($fetch['status'] >= 200 && $fetch['status'] < 300);
    $finalUrlDiffers   = ($fetch['final_url'] !== '' && rtrim($fetch['final_url'], '/') !== rtrim($url, '/'));
    $canonicalStatus   = 'Missing Canonical';
    if ($canonical !== '') {
        $normCanon = rtrim($canonical, '/');
        $normUrl   = rtrim($url, '/');
        $canonicalStatus = ($normCanon === $normUrl) ? 'Self Canonical' : 'Canonical To Different URL';
    }
    $indexabilityStatus = 'Indexable';
    $indexabilityReason = '';
    if ($noindex) {
        $indexabilityStatus = 'Not Indexable';
        $indexabilityReason = 'noindex directive found';
    } elseif (!$statusIndexable) {
        $indexabilityStatus = 'Not Indexable';
        $indexabilityReason = 'HTTP status ' . $fetch['status'];
    } elseif ($canonicalStatus === 'Canonical To Different URL') {
        $indexabilityStatus = 'Maybe';
        $indexabilityReason = 'Canonical points to a different URL';
    }

    $discoveredLinks         = leAuditExtractLinks($html, $url, $config);
    $discoveredExternalLinks = leAuditExtractExternalLinks($html, $config);
    $discoveredImages        = leAuditExtractImages($html, $url, $config);
    $discoveredLinkContexts  = leAuditExtractLinkContexts($html, $url, $config);
    $discoveredImageContexts = leAuditExtractImageContexts($html, $url, $config);

    return [
        'url'                     => $url,
        'status'                  => $fetch['status'],
        'final_url'               => $fetch['final_url'],
        'load_time_ms'            => $fetch['load_time_ms'],
        'title'                   => $title,
        'title_length'            => $titleLen,
        'meta_description'        => $description,
        'meta_description_length' => $descLen,
        'canonical'               => $canonical,
        'h1_count'                => $h1Count,
        'h1_text'                 => $h1Text,
        'noindex'                 => $noindex,
        'images_without_alt'      => $imgsNoAlt,
        'issues'                  => $issues,
        'warnings'                => $warnings,
        'scanned_at'              => gmdate('c'),
        // Performance diagnostics
        'html_size_kb'            => $htmlSizeKb,
        'script_count'            => $scriptCount,
        'stylesheet_count'        => $stylesheetCount,
        'image_count'             => $imageCount,
        'inline_script_count'     => $inlineScriptCount,
        'third_party_domain_count'=> $thirdPartyDomainCount,
        'slow_tier'               => $slowTier,
        // Content quality
        'word_count'              => $wordCount,
        'thin_content'            => $thinContent,
        'h2_count'                => $h2Count,
        'h3_count'                => $h3Count,
        'paragraph_count'         => $paragraphCount,
        // Structured data
        'json_ld_count'           => $jsonLdCount,
        'invalid_json_ld_count'   => $invalidJsonLdCount,
        'schema_types'            => $schemaTypes,
        // Page type and importance
        'page_type'               => $pageType,
        'is_important'            => $isImportant,
        'importance_score'        => $importanceScore,
        // Indexability
        'indexability_status'     => $indexabilityStatus,
        'indexability_reason'     => $indexabilityReason,
        'canonical_status'        => $canonicalStatus,
        'final_url_differs'       => $finalUrlDiffers,
        // Links (populated below)
        'discovered_links'          => $discoveredLinks,
        'discovered_external_links' => $discoveredExternalLinks,
        'discovered_images'         => $discoveredImages,
        'discovered_link_contexts'  => $discoveredLinkContexts,
        'discovered_image_contexts' => $discoveredImageContexts,
    ];
}

function leAuditLoadSitemapUrls(array $config): array
{
    $urls           = [];
    $sitemapsToRead = [];
    $sitemapsRead   = [];
    $diagnostics    = [];

        $__leForeachTmp11 = $config['sitemap_candidates'];
    $__leForeachKeys11 = is_array($__leForeachTmp11) ? array_keys($__leForeachTmp11) : [];
    for ($__leForeachIndex11 = 0, $__leForeachCount11 = count($__leForeachKeys11); $__leForeachIndex11 < $__leForeachCount11; $__leForeachIndex11++) {
        $__leForeachKey11 = $__leForeachKeys11[$__leForeachIndex11];
        $sitemapUrl = $__leForeachTmp11[$__leForeachKey11];
        $sitemapUrl = trim((string) $sitemapUrl);
        if ($sitemapUrl !== '') {
            $sitemapsToRead[] = $sitemapUrl;
        }
        }

    while (!empty($sitemapsToRead)) {
        $currentSitemap = array_shift($sitemapsToRead);

        if ($currentSitemap === '' || isset($sitemapsRead[$currentSitemap])) {
            continue;
        }

        $sitemapsRead[$currentSitemap] = true;
        $fetch = leAuditFetchUrl($currentSitemap, $config, $config['sitemap_timeout']);

        if ($fetch['status'] < 200 || $fetch['status'] >= 300 || $fetch['body'] === '') {
            $errDetail     = $fetch['error'] !== '' ? ': ' . $fetch['error'] : '';
            $diagnostics[] = 'SKIP ' . $currentSitemap . ' (HTTP ' . $fetch['status'] . $errDetail . ')';
            continue;
        }

        $body = $fetch['body'];

        if (!preg_match_all('#<loc>(.*?)</loc>#is', $body, $matches)) {
            $diagnostics[] = 'SKIP ' . $currentSitemap . ' (no <loc> entries found)';
            continue;
        }

        // Log a sample of the first 3 raw <loc> values to aid debugging
        $sampleLocs = array_slice($matches[1], 0, 3);
                $__leForeachTmp12 = $sampleLocs;
        $__leForeachKeys12 = is_array($__leForeachTmp12) ? array_keys($__leForeachTmp12) : [];
        for ($__leForeachIndex12 = 0, $__leForeachCount12 = count($__leForeachKeys12); $__leForeachIndex12 < $__leForeachCount12; $__leForeachIndex12++) {
            $__leForeachKey12 = $__leForeachKeys12[$__leForeachIndex12];
            $s =& $__leForeachTmp12[$__leForeachKey12];
            $s = trim(html_entity_decode(strip_tags($s), ENT_QUOTES, 'UTF-8'));
                }
        unset($s);
        $diagnostics[] = 'SAMPLE locs from ' . $currentSitemap . ': ' . implode(' | ', $sampleLocs);

        $isSitemapIndex = stripos($body, '<sitemapindex') !== false
                       || (stripos($body, '<sitemap>') !== false && stripos($body, '<url>') === false);
        $foundUnique    = [];
        $subSitemaps    = 0;

                $__leForeachTmp13 = $matches[1];
        $__leForeachKeys13 = is_array($__leForeachTmp13) ? array_keys($__leForeachTmp13) : [];
        for ($__leForeachIndex13 = 0, $__leForeachCount13 = count($__leForeachKeys13); $__leForeachIndex13 < $__leForeachCount13; $__leForeachIndex13++) {
            $__leForeachKey13 = $__leForeachKeys13[$__leForeachIndex13];
            $loc = $__leForeachTmp13[$__leForeachKey13];
            $loc = trim(html_entity_decode(strip_tags($loc), ENT_QUOTES, 'UTF-8'));

            if ($loc === '') {
                continue;
            }

            if ($isSitemapIndex) {
                if (!isset($sitemapsRead[$loc])) {
                    $sitemapsToRead[] = $loc;
                    $subSitemaps++;
                }
                continue;
            }

            // Treat any <loc> pointing to an .xml file on an allowed host as a sub-sitemap
            $locHost = strtolower((string) parse_url($loc, PHP_URL_HOST));
            $locPath = (string) parse_url($loc, PHP_URL_PATH);
            if (in_array($locHost, $config['allowed_hosts'], true) && preg_match('#\.xml$#i', $locPath)) {
                if (!isset($sitemapsRead[$loc])) {
                    $sitemapsToRead[] = $loc;
                    $subSitemaps++;
                }
                continue;
            }

            $normalized = leAuditNormalizeUrl($loc, $config);
            if ($normalized !== '') {
                $urls[]                    = $normalized;
                $foundUnique[$normalized]  = true;
            }
                }

        if ($isSitemapIndex) {
            $diagnostics[] = 'INDEX ' . $currentSitemap . ' -> queued ' . $subSitemaps . ' sub-sitemaps';
        } elseif ($subSitemaps > 0) {
            $diagnostics[] = 'MIXED ' . $currentSitemap . ' -> ' . count($foundUnique) . ' page URLs + queued ' . $subSitemaps . ' sub-sitemaps';
        } else {
            $foundUrls     = count($matches[1]);
            $diagnostics[] = 'OK ' . $currentSitemap . ' -> ' . count($foundUnique) . ' unique page URLs (' . $foundUrls . ' raw)';
        }
    }

    return [
        'urls'        => array_values(array_unique($urls)),
        'diagnostics' => $diagnostics,
    ];
}

function leAuditAddUrlsToQueue(array &$queue, array $urls, array $config, string $sourceUrl = ''): int
{
    $added = 0;

    if (!isset($queue['referrers'])) {
        $queue['referrers'] = [];
    }

        $__leForeachTmp14 = $urls;
    $__leForeachKeys14 = is_array($__leForeachTmp14) ? array_keys($__leForeachTmp14) : [];
    for ($__leForeachIndex14 = 0, $__leForeachCount14 = count($__leForeachKeys14); $__leForeachIndex14 < $__leForeachCount14; $__leForeachIndex14++) {
        $__leForeachKey14 = $__leForeachKeys14[$__leForeachIndex14];
        $url = $__leForeachTmp14[$__leForeachKey14];
        $normalized = leAuditNormalizeUrl($url, $config);

        if ($normalized === '') {
            continue;
        }

        if (isset($queue['seen'][$normalized]) || isset($queue['scanned'][$normalized])) {
            continue;
        }

        if (count($queue['seen']) >= $config['max_queue_size']) {
            break;
        }

        $queue['pending'][]         = $normalized;
        $queue['seen'][$normalized] = true;

        if ($sourceUrl !== '' && !isset($queue['referrers'][$normalized])) {
            $queue['referrers'][$normalized] = $sourceUrl;
        }

        $added++;
        }

    $queue['pending']    = array_values(array_unique($queue['pending']));
    $queue['updated_at'] = gmdate('c');
    return $added;
}

function leAuditBuildSummary(array $results): array
{
    $summary = [
        'total_scanned'        => 0,
        'critical_pages'       => 0,
        'warning_pages'        => 0,
        'passed_pages'         => 0,
        'not_found_pages'      => 0,
        'server_error_pages'   => 0,
        'redirect_pages'       => 0,
        'missing_titles'       => 0,
        'missing_descriptions' => 0,
        'missing_canonicals'   => 0,
        'noindex_pages'        => 0,
        'slow_pages'           => 0,
        'images_missing_alt'   => 0,
        'avg_load_time_ms'     => 0,
        'broken_links_total'   => !empty($results['broken_links']) ? count($results['broken_links']) : 0,
    ];

    if (empty($results['items']) || !is_array($results['items'])) {
        return $summary;
    }

    $totalLoadTime = 0;

        $__leForeachTmp15 = $results['items'];
    $__leForeachKeys15 = is_array($__leForeachTmp15) ? array_keys($__leForeachTmp15) : [];
    for ($__leForeachIndex15 = 0, $__leForeachCount15 = count($__leForeachKeys15); $__leForeachIndex15 < $__leForeachCount15; $__leForeachIndex15++) {
        $__leForeachKey15 = $__leForeachKeys15[$__leForeachIndex15];
        $item = $__leForeachTmp15[$__leForeachKey15];
        $summary['total_scanned']++;

        if (!empty($item['issues'])) {
            $summary['critical_pages']++;
        } elseif (!empty($item['warnings'])) {
            $summary['warning_pages']++;
        } else {
            $summary['passed_pages']++;
        }

        if ((int) $item['status'] === 404) {
            $summary['not_found_pages']++;
        }
        if ((int) $item['status'] >= 500) {
            $summary['server_error_pages']++;
        }
        if ((int) $item['status'] >= 300 && (int) $item['status'] < 400) {
            $summary['redirect_pages']++;
        }
        if (trim((string) $item['title']) === '') {
            $summary['missing_titles']++;
        }
        if (trim((string) $item['meta_description']) === '') {
            $summary['missing_descriptions']++;
        }
        if (trim((string) $item['canonical']) === '') {
            $summary['missing_canonicals']++;
        }
        if (!empty($item['noindex'])) {
            $summary['noindex_pages']++;
        }
        if ((int) $item['load_time_ms'] > 3000) {
            $summary['slow_pages']++;
        }
        if (!empty($item['images_without_alt']) && (int) $item['images_without_alt'] > 0) {
            $summary['images_missing_alt']++;
        }

        $totalLoadTime += (int) $item['load_time_ms'];
        }

    if ($summary['total_scanned'] > 0) {
        $summary['avg_load_time_ms'] = (int) round($totalLoadTime / $summary['total_scanned']);
    }

    // Add pages that were scanned but not stored (store_passed_pages = false)
    $extraPassed = (int) ($results['passed_pages_count'] ?? 0);
    $summary['passed_pages']  += $extraPassed;
    $summary['total_scanned'] += $extraPassed;

    return $summary;
}

function leAuditMergeBrokenContext(array $brokenLink, array $context): array
{
    $brokenLink['link_text']      = $context['link_text']      ?? '';
    $brokenLink['raw_href']       = $context['raw_href']       ?? '';
    $brokenLink['source_excerpt'] = $context['source_excerpt'] ?? '';
    $brokenLink['html_snippet']   = $context['html_snippet']   ?? '';
    return $brokenLink;
}

function leAuditFormatEta(int $seconds): string
{
    if ($seconds <= 0) {
        return 'calculating';
    }

    $hours = intdiv($seconds, 3600);
    $minutes = intdiv($seconds % 3600, 60);
    $secs = $seconds % 60;

    if ($hours > 0) {
        return $hours . 'h ' . $minutes . 'm';
    }

    if ($minutes > 0) {
        return $minutes . 'm ' . $secs . 's';
    }

    return $secs . 's';
}

function leAuditExportCsv(array $results): void
{
    // Always clean the output buffer first so no Joomla HTML leaks before the file.
    while (ob_get_level() > 0) {
        ob_end_clean();
    }

    $filename = 'lottoexpert-full-site-audit-' . gmdate('Ymd-His') . '.csv';
    $csvHeaders = [
        'Record Type', 'URL', 'Status', 'Load Time MS', 'Title', 'Title Length',
        'Meta Description', 'Meta Description Length', 'Canonical',
        'H1 Count', 'H1 Text', 'Noindex', 'Images Without Alt',
        'Issues', 'Warnings', 'Scanned At',
        'Broken Link Type', 'Source Page', 'Broken URL', 'Broken Status',
        'Link Text Or Image Alt', 'Raw Href Or Src', 'Page Text Near Link', 'HTML Snippet', 'Detected At',
    ];

    if (!headers_sent()) {
        header('Content-Type: text/csv; charset=UTF-8');
        header('Content-Disposition: attachment; filename="' . $filename . '"');
        header('Cache-Control: no-store, no-cache, must-revalidate, max-age=0');
        header('Pragma: no-cache');
        header('Expires: 0');
    }

    $fp = fopen('php://output', 'w');
    fwrite($fp, "\xEF\xBB\xBF");
    fputcsv($fp, $csvHeaders);

    if (!empty($results['items'])) {
                $__leForeachTmp16 = $results['items'];
        $__leForeachKeys16 = is_array($__leForeachTmp16) ? array_keys($__leForeachTmp16) : [];
        for ($__leForeachIndex16 = 0, $__leForeachCount16 = count($__leForeachKeys16); $__leForeachIndex16 < $__leForeachCount16; $__leForeachIndex16++) {
            $__leForeachKey16 = $__leForeachKeys16[$__leForeachIndex16];
            $item = $__leForeachTmp16[$__leForeachKey16];
            fputcsv($fp, [
                'page',
                $item['url']                      ?? '',

    if (!empty($results['broken_links'])) {

}

// -- Page Type Detection ------------------------------------------------------
function leAuditDetectPageType(string $url, string $title, string $h1): string
{
    $u = strtolower($url);

    // Homepage
    if (preg_match('#^https?://[^/]+(/?|/index\.php/?)(\?.*)?$#i', $url)) {
        return 'Homepage';
    }

    // Utility/system
    if (preg_match('#(print=|tmpl=component|format=feed|/component/|/administrator/)#i', $url)) {
        return 'Utility Page';
    }

    // Account/checkout
    if (preg_match('#/(account|login|register|checkout|cart|my-profile|membership/join)#i', $url)) {
        return 'Checkout Account Page';
    }

    // Membership
    if (preg_match('#/(membership|subscribe|subscription|premium|upgrade)#i', $url)) {
        return 'Membership Page';
    }

    // Winning numbers (check before generic lottery)
    if (preg_match('#/(winning-numbers|lottery-results|results|past-results|winning-result)#i', $url)) {
        return 'Winning Numbers Page';
    }

    // Prediction / tool
    if (preg_match('#/(prediction|predictions|predict|hot-numbers|frequency|due-numbers)#i', $url)) {
        return 'Prediction Tool Page';
    }

    // Analysis / tool / scanner
    if (preg_match('#/(analysis|analyzer|analyser|tool|scanner|calculator|strategy|guide|how-to-win)#i', $url)) {
        return 'Analysis Tool Page';
    }

    // Archive
    if (preg_match('#/(archive|archives)/|/\d{4}/\d{2}/#i', $url)) {
        return 'Archive Page';
    }

    // Blog / article / news
    if (preg_match('#/(blog|article|articles|news|post|posts)/|/news-#i', $url)) {
        return 'Blog Article';
    }

    // State lottery pages (powerball, mega millions, state lotteries)
    if (preg_match('#/(powerball|mega-millions|megamillions|pick-3|pick-4|pick3|pick4|daily-3|daily-4|fantasy-5|fantasy5|cash4life|lotto|cash-pop|lucky-for-life|all-or-nothing|keno|hot-lotto|classic-lotto|florida-lottery|texas-lottery|california-lottery|new-york-lottery|illinois-lottery|georgia-lottery|ohio-lottery|pennsylvania-lottery|michigan-lottery|new-jersey-lottery|north-carolina-lottery|virginia-lottery|indiana-lottery|tennessee-lottery|colorado-lottery|washington-lottery|maryland-lottery|iowa-lottery|kansas-lottery|missouri-lottery|multi-state|multistate)#i', $url)) {
        return 'State Lottery Page';
    }

    // Fallback on title / h1
    $lowerTitle = strtolower($title . ' ' . $h1);
    if (strpos($lowerTitle, 'winning numbers') !== false || strpos($lowerTitle, 'lottery results') !== false) {
        return 'Winning Numbers Page';
    }
    if (strpos($lowerTitle, 'prediction') !== false || strpos($lowerTitle, 'hot numbers') !== false) {
        return 'Prediction Tool Page';
    }
    if (strpos($lowerTitle, 'powerball') !== false || strpos($lowerTitle, 'mega millions') !== false || strpos($lowerTitle, 'lottery') !== false) {
        return 'State Lottery Page';
    }

    return 'Unknown';
}

// -- Importance Detection -----------------------------------------------------
function leAuditDetectImportance(string $url, string $pageType): array
{
    $scoreMap = [
        'Homepage'             => 100,
        'State Lottery Page'   => 90,
        'Prediction Tool Page' => 88,
        'Winning Numbers Page' => 85,
        'Membership Page'      => 80,
        'Analysis Tool Page'   => 75,
        'Checkout Account Page'=> 60,
        'Blog Article'         => 40,
        'Unknown'              => 35,
        'Archive Page'         => 20,
        'Utility Page'         => 5,
    ];
    $score      = $scoreMap[$pageType] ?? 35;
    $isImportant = $score >= 60;
    return ['is_important' => $isImportant, 'importance_score' => $score];
}

// -- Site Health Scoring ------------------------------------------------------
function leAuditComputeHealthScores(array $results, array $summary): array
{
    $total = max(1, (int) $summary['total_scanned']);

    $seo           = 100;
    $crawlability  = 100;
    $performance   = 100;
    $accessibility = 100;
    $indexability  = 100;
    $brokenLink    = 100;

    $seo          -= min(40, (int) $summary['missing_titles']       * 2);
    $seo          -= min(20, (int) $summary['missing_descriptions']  * 1);
    $seo          -= min(10, (int) $summary['missing_canonicals']    * 1);
    $crawlability -= min(40, (int) $summary['not_found_pages']       * 3);
    $crawlability -= min(30, (int) $summary['server_error_pages']    * 5);
    $performance  -= min(30, (int) $summary['slow_pages']            * 2);
    $accessibility-= min(40, (int) $summary['images_missing_alt']    * 2);
    $indexability -= min(40, (int) $summary['noindex_pages']         * 3);
    $brokenLink   -= min(50, (int) $summary['broken_links_total']    * 2);

    $seo          = max(0, $seo);
    $crawlability = max(0, $crawlability);
    $performance  = max(0, $performance);
    $accessibility= max(0, $accessibility);
    $indexability = max(0, $indexability);
    $brokenLink   = max(0, $brokenLink);
    $overall      = (int) round(($seo + $crawlability + $performance + $accessibility + $indexability + $brokenLink) / 6);

    return [
        'overall'       => $overall,
        'seo'           => $seo,
        'crawlability'  => $crawlability,
        'performance'   => $performance,
        'accessibility' => $accessibility,
        'indexability'  => $indexability,
        'broken_link'   => $brokenLink,
    ];
}

// -- Additional Export Functions -----------------------------------------------
function leAuditStartExport(string $filename, string $contentType): void
{
    // Always clean OB first so buffered Joomla/template HTML never leaks into the download.
    while (ob_get_level() > 0) { ob_end_clean(); }
    if (!headers_sent()) {
        header('Content-Type: ' . $contentType . '; charset=UTF-8');
        header('Content-Disposition: attachment; filename="' . $filename . '"');
        header('Cache-Control: no-store, no-cache, must-revalidate, max-age=0');
        header('Pragma: no-cache');
        header('Expires: 0');
    }
}

    $fp = fopen('php://output', 'w');
    fwrite($fp, "\xEF\xBB\xBF");
    fputcsv($fp, ['URL', 'Status', 'Page Type', 'Important', 'Images Without Alt', 'Accessibility Warnings', 'Title', 'Scanned At']);
        $__leForeachTmp26 = $results['items'] ?? [];
    $__leForeachKeys26 = is_array($__leForeachTmp26) ? array_keys($__leForeachTmp26) : [];
    for ($__leForeachIndex26 = 0, $__leForeachCount26 = count($__leForeachKeys26); $__leForeachIndex26 < $__leForeachCount26; $__leForeachIndex26++) {
        $__leForeachKey26 = $__leForeachKeys26[$__leForeachIndex26];

        $accWarnings = [];
                $__leForeachTmp2 = $item['warnings'] ?? [];

        for ($__leForeachIndex2 = 0, $__leForeachCount2 = count($__leForeachKeys2); $__leForeachIndex2 < $__leForeachCount2; $__leForeachIndex2++) {

            if (strpos($wl, 'alt') !== false || strpos($wl, 'accessibility') !== false

$message = '';

    leAuditSessionSet('le_queue', $queue);

} elseif ($action === 'discover_sitemap') {
    $sitemapResult = leAuditLoadSitemapUrls($config);
    $sitemapUrls   = $sitemapResult['urls'];
    $diagnostics   = $sitemapResult['diagnostics'];
    $added         = leAuditAddUrlsToQueue($queue, $sitemapUrls, $config);

    $alreadyQueued = count($sitemapUrls) - $added;
    $diagText      = !empty($diagnostics) ? ' Details: ' . implode(' | ', $diagnostics) : '';
    $message = 'Sitemap discovery complete. Found ' . count($sitemapUrls) . ' page URLs. '
             . 'Added ' . (int) $added . ' new URLs to queue'
             . ($alreadyQueued > 0 ? ' (' . $alreadyQueued . ' already queued).' : '.') . $diagText;
} elseif ($action === 'scan_batch' || $action === 'scan_batch_ajax') {
    @set_time_limit(180);
    @ignore_user_abort(true);

    // Re-read scan options from the embedded batch form fields and re-persist them.
    // This ensures settings survive across batches even if the Joomla session is
    // regenerated between requests (common in Sourcerer-embedded articles).
    $optionKeys = array_keys(leAuditGetDefaultOptions());
    if ($input->getInt('batch_opt_check_seo', -1) !== -1) {
                $__leForeachTmp28 = $optionKeys;
        $__leForeachKeys28 = is_array($__leForeachTmp28) ? array_keys($__leForeachTmp28) : [];

    $processed    = 0;
    $newUrlsFound = 0;
    $batchLimit   = (int) $config['batch_limit'];

    if (!isset($queue['ext_pending']))        { $queue['ext_pending']        = []; }
    if (!isset($queue['ext_seen']))           { $queue['ext_seen']           = []; }
    if (!isset($queue['ext_referrers']))      { $queue['ext_referrers']      = []; }

    if (!isset($queue['int_link_seen']))      { $queue['int_link_seen']      = []; }

    if (!isset($queue['int_link_contexts']))  { $queue['int_link_contexts']  = []; }

    $batchStartedAt = microtime(true);

    while ($processed < $batchLimit && !empty($queue['pending'])) {
        $url        = array_shift($queue['pending']);
        $normalized = leAuditNormalizeUrl($url, $config);

        if ($normalized === '') {
            continue;
        }

        if (isset($queue['scanned'][$normalized])) {
            continue;
        }

        $result              = leAuditCheckUrl($normalized, $config, $scanOptions);
        $discoveredLinks         = $result['discovered_links'];
        $discoveredExtLinks      = $result['discovered_external_links'] ?? [];
        $discoveredImages        = $result['discovered_images'] ?? [];
        $discoveredLinkContexts  = $result['discovered_link_contexts'] ?? [];
        $discoveredImageContexts = $result['discovered_image_contexts'] ?? [];
        unset($result['discovered_links'], $result['discovered_external_links'], $result['discovered_images'], $result['discovered_link_contexts'], $result['discovered_image_contexts']);

        $hasIssues   = !empty($result['issues']);
        $hasWarnings = !empty($result['warnings']);
        $isPassing   = !$hasIssues && !$hasWarnings;

        // Optionally skip storing passed items to reduce session size

            $results['passed_pages_count'] = ($results['passed_pages_count'] ?? 0) + 1;
        } else {
            $result['source_page'] = $queue['referrers'][$normalized] ?? '';
            $results['items'][$normalized] = $result;
        }
        $queue['scanned'][$normalized] = true;

        $newUrlsFound += leAuditAddUrlsToQueue($queue, $discoveredLinks, $config, $normalized);
                $__leForeachTmp29 = $discoveredLinks;
        $__leForeachKeys29 = is_array($__leForeachTmp29) ? array_keys($__leForeachTmp29) : [];
        for ($__leForeachIndex29 = 0, $__leForeachCount29 = count($__leForeachKeys29); $__leForeachIndex29 < $__leForeachCount29; $__leForeachIndex29++) {
            $__leForeachKey29 = $__leForeachKeys29[$__leForeachIndex29];
            $contextUrl = $__leForeachTmp29[$__leForeachKey29];
            if (isset($discoveredLinkContexts[$contextUrl]) && !isset($queue['link_contexts'][$contextUrl])) {
                $queue['link_contexts'][$contextUrl] = $discoveredLinkContexts[$contextUrl];
            }
                }

        // Queue external links and images for HEAD-check (only if those checks are enabled)
        $extAssets = [];
        if ($scanOptions['check_broken_external'] ?? true) {
            $extAssets = array_merge($extAssets, $discoveredExtLinks);
        }
        if ($scanOptions['check_broken_images'] ?? true) {
            $extAssets = array_merge($extAssets, $discoveredImages);
        }
                $__leForeachTmp30 = $extAssets;
        $__leForeachKeys30 = is_array($__leForeachTmp30) ? array_keys($__leForeachTmp30) : [];
        for ($__leForeachIndex30 = 0, $__leForeachCount30 = count($__leForeachKeys30); $__leForeachIndex30 < $__leForeachCount30; $__leForeachIndex30++) {
            $__leForeachKey30 = $__leForeachKeys30[$__leForeachIndex30];
            $extUrl = $__leForeachTmp30[$__leForeachKey30];
            if (!isset($queue['ext_seen'][$extUrl]) && count($queue['ext_seen']) < 10000) {
                $queue['ext_pending'][]          = $extUrl;
                $queue['ext_seen'][$extUrl]      = true;
                $queue['ext_referrers'][$extUrl] = $normalized;
                if (isset($discoveredImageContexts[$extUrl])) {
                    $queue['ext_contexts'][$extUrl] = $discoveredImageContexts[$extUrl];
                } elseif (isset($discoveredLinkContexts[$extUrl])) {
                    $queue['ext_contexts'][$extUrl] = $discoveredLinkContexts[$extUrl];
                }
            }
                }

        // Queue internal links for dedicated HEAD-check (proactive; separate from main crawl)
        if ($scanOptions['check_broken_internal_links'] ?? true) {
                        $__leForeachTmp31 = $discoveredLinks;
            $__leForeachKeys31 = is_array($__leForeachTmp31) ? array_keys($__leForeachTmp31) : [];
            for ($__leForeachIndex31 = 0, $__leForeachCount31 = count($__leForeachKeys31); $__leForeachIndex31 < $__leForeachCount31; $__leForeachIndex31++) {
                $__leForeachKey31 = $__leForeachKeys31[$__leForeachIndex31];
                $intLinkUrl = $__leForeachTmp31[$__leForeachKey31];
                if (!isset($queue['int_link_seen'][$intLinkUrl]) && count($queue['int_link_seen']) < 20000) {
                    $queue['int_link_pending'][]              = $intLinkUrl;
                    $queue['int_link_seen'][$intLinkUrl]      = true;
                    $queue['int_link_referrers'][$intLinkUrl] = $normalized;
                    if (isset($discoveredLinkContexts[$intLinkUrl])) {

                        $results['broken_links'][$blKey]['last_seen_at'] = gmdate('c');

                        $results['broken_links'][$blKey] = leAuditMergeBrokenContext([
                            'source_page'      => $referrer,

                        'found_at'         => gmdate('c'),
                        'first_seen_at'    => gmdate('c'),
                        'last_seen_at'     => gmdate('c'),
                        'occurrence_count' => 1,
                        'link_type'        => $linkType,
                        'check_method'     => $extResult['method'],
                    ], $queue['ext_contexts'][$extUrl] ?? []);
                }
            }
        }
        $extChecked++;
    }

    // HEAD-check internal links found on pages (up to 10 per batch; separate from main crawl)
    $intLinkChecked = 0;
    while ($intLinkChecked < 10 && !empty($queue['int_link_pending'])) {

        // Skip URLs already fully crawled - check_broken_internal handles those
        if (isset($queue['scanned'][$intLinkUrl])) {

        }

        $intLinkResult = leAuditHeadCheckUrl($intLinkUrl);
        $intLinkStatus = $intLinkResult['status'];
        $isBroken      = ($intLinkStatus === 0 || $intLinkStatus === 404

                    $results['broken_links'][$blKey]['occurrence_count']++;

                        'source_page'      => $intLinkReferrer,
                        'broken_url'       => $intLinkUrl,
                        'status'           => $intLinkStatus,
                        'found_at'         => gmdate('c'),
                        'first_seen_at'    => gmdate('c'),
                        'last_seen_at'     => gmdate('c'),
                        'occurrence_count' => 1,
                        'link_type'        => 'internal',
                        'check_method'     => $intLinkResult['method'],
                    ], $queue['int_link_contexts'][$intLinkUrl] ?? []);

            }
        }

    $maxItems = (int) ($config['max_items_stored']        ?? 5000);
    $maxBL    = (int) ($config['max_broken_links_stored'] ?? 1000);
    if (count($results['items']) > $maxItems) {
        $results['items'] = array_slice($results['items'], -$maxItems, null, true);
    }
    if (!empty($results['broken_links']) && count($results['broken_links']) > $maxBL) {
        $results['broken_links'] = array_slice($results['broken_links'], -$maxBL, null, true);
    }

    $results['summary']    = leAuditBuildSummary($results);
    $results['updated_at'] = gmdate('c');
    $queue['updated_at']   = gmdate('c');
    leAuditSessionSet('le_queue', $queue);
    leAuditSessionSet('le_results', $results);

    $pendingAfter       = count($queue['pending']);
    $extPendingAfter    = count($queue['ext_pending'] ?? []);
    $intLinkPendingAfter = count($queue['int_link_pending'] ?? []);
    $message = 'Batch complete. Scanned ' . (int) $processed . ' pages. '
             . (int) $newUrlsFound . ' new pages discovered. '
             . (int) $pendingAfter . ' pages still pending. '
             . (int) $extChecked . ' external asset(s) checked, '
             . (int) $extPendingAfter . ' still queued. '
             . (int) $intLinkChecked . ' internal link(s) HEAD-checked, '
             . (int) $intLinkPendingAfter . ' still queued.';

    if ($action === 'scan_batch_ajax') {
        $ajaxPendingCount = count($queue['pending'] ?? []);
        $ajaxSeenCount    = count($queue['seen']    ?? []);

        try {
            if (typeof fd['delete'] === 'function') { fd['delete'](name); }
        } catch (e) { /* intentional: append fallback handles duplicates */ }
        fd.append(name, value);
    }

        }
    }

        }

    function leRunAjaxBatch() {
        if (sessionStorage.getItem(AUTORUN_KEY) !== '1') { return; }
        var ajaxForm = document.getElementById('leAutoScanForm');
        if (!ajaxForm) { return; }
        var opts = leLoadOpts();
        var fd   = new FormData(ajaxForm);
        leFormDataSet(fd, 'audit_action', 'scan_batch_ajax');
        var leOptKeys = Object.keys(opts);
        for (var leOptKi = 0; leOptKi < leOptKeys.length; leOptKi++) {
            leFormDataSet(fd, 'batch_opt_' + leOptKeys[leOptKi], opts[leOptKeys[leOptKi]] ? '1' : '0');
        }
        var leXhr2 = new XMLHttpRequest();
        leXhr2.open('POST', ajaxForm.action, true);
        leXhr2.onreadystatechange = function () {
            if (leXhr2.readyState !== 4) { return; }
            if (leXhr2.status >= 200 && leXhr2.status < 300) {
                var data = null;
                try { data = JSON.parse(leXhr2.responseText); } catch (e) {}
                if (!data) {
                    // JSON parse failed - treat as error, wait and retry
                    if (sessionStorage.getItem(AUTORUN_KEY) === '1') {
                        setTimeout(leRunAjaxBatch, 3000);
                    }
                    return;
                }
                leUpdateProgress(data.progress_pct, data.scanned, data.seen, data.pending, data.eta_label);
                if (pendingSpan) { pendingSpan.textContent = data.pending; }
                if (sessionStorage.getItem(AUTORUN_KEY) !== '1') { return; }
                if (data.done || data.pending <= 0) {
                    sessionStorage.removeItem(AUTORUN_KEY);
                    sessionStorage.removeItem(SCROLL_KEY);
                    // Reload the page so PHP re-renders results from session and
                    // the export button is available. The on-load pending===0 branch
                    // will show the "done" banner automatically.
                    window.location.reload();
                    return;
                }
                setTimeout(leRunAjaxBatch, 100);
            } else {
                // On any error (504, network failure) wait and retry
                if (sessionStorage.getItem(AUTORUN_KEY) === '1') {
                    setTimeout(leRunAjaxBatch, 3000);
                }
            }
        };
        leXhr2.send(fd);
    }

    function leFilterTable(tableId, filterFn) {
        var table = document.getElementById(tableId);
        if (!table) { return; }
        var tbody = table.getElementsByTagName('tbody')[0];
        if (!tbody) { return; }
        var rows = tbody.getElementsByTagName('tr');
        for (var i = 0; i < rows.length; i++) {
            rows[i].style.display = filterFn(rows[i]) ? '' : 'none';
        }
    }

    function leApplyCritFilter() {
        var urlVal  = (document.getElementById('leCritFilterUrl')       || {}).value  || '';
        var statVal = (document.getElementById('leCritFilterStatus')    || {}).value  || '';
        var impEl   = document.getElementById('leCritFilterImportant');
        var impOnly = impEl ? impEl.checked : false;
        urlVal  = urlVal.toLowerCase();
        leFilterTable('leCritTable', function (row) {
            var rowUrl    = (row.getAttribute('data-url')       || '').toLowerCase();
            var rowStatus = row.getAttribute('data-status')     || '';
            var rowImp    = row.getAttribute('data-important')  || '0';
            if (urlVal   && rowUrl.indexOf(urlVal) === -1)   { return false; }
            if (statVal  && rowStatus !== statVal)            { return false; }
            if (impOnly  && rowImp !== '1')                   { return false; }
            return true;
        });
    }

    var critImpCheck   = document.getElementById('leCritFilterImportant');
    if (critUrlInput)   { critUrlInput.addEventListener('input',  leApplyCritFilter); }
    if (critStatSelect) { critStatSelect.addEventListener('change', leApplyCritFilter); }
    if (critImpCheck)   { critImpCheck.addEventListener('change',  leApplyCritFilter); }
